\section{\EN{Passing arguments}\RU{Передача аргументов}}

\EN{Let's extend our \TT{min()}/\TT{max()} example:}
\RU{Теперь расширим пример \TT{min()}/\TT{max()}:}

\begin{lstlisting}
public class minmax
{
	public static int min (int a, int b)
	{
		if (a>b)
			return b;
		return a;
	}

	public static int max (int a, int b)
	{
		if (a>b)
			return a;
		return b;
	}

	public static void main(String[] args)
	{
		int a=123, b=456;
		int max_value=max(a, b);
		int min_value=min(a, b);
		System.out.println(min_value);
		System.out.println(max_value);
	}
}
\end{lstlisting}
