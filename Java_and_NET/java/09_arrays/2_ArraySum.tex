\subsection{\EN{Summing elements of array}\RU{Суммирование элементов массива}}

\EN{Another example}\RU{Еще один пример}:

\begin{lstlisting}
public class ArraySum
{
	public static int f (int[] a)
	{
		int sum=0;
		for (int i=0; i<a.length; i++)
			sum=sum+a[i];
		return sum;
	}
}
\end{lstlisting}
